DrawText()
-   creates triangles
-   takes the font textures and renders those triangles (same as drawing player)
-   is NOT a print statement
-   should be used in Render()
    -   because it's rendering something
-   DO NOT USE in Update()


----------------

void Render() {
    glClear(GL_COLOR_BUFFER_BIT);

    // render and display all platforms
    for (size_t i = 0; i < PLATFORM_COUNT; i++) {
        state.platforms[i].Render(&program);
    }


    state.player->Render(&program);

    SDL_GL_SwapWindow(displayWindow);
}
