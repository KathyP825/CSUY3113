Current Timestep Code:
-   is as fast as our computer can go
-   is variable (time changes everytime)
-   looking for time between frames and moving based on that difference in time

void Update() {
    float ticks = (float)SDL_GetTicks() / 1000.0f;
    float deltaTime = ticks - lastTicks;
    lastTicks = ticks;

    // Add (direction * units per second * elapsed time)
    player_position += player_movement * player_speed * deltaTime;
}
